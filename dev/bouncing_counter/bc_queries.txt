BTWN = (eq c c0()  || lt c0() c) && (eq c c100() || lt c c100());

!(eq c c0() && !BTWN);
!(eq c c100() && eq c c0());
eq c c0()    && [c <- add c c1()] -> X BTWN;
eq c c100()  && [c <- sub c c1()] -> X BTWN;
!(eq c c0())   && BTWN && [c <- sub c c1()] -> X BTWN;
!(eq c c100()) && BTWN && [c <- add c c1()] -> X BTWN;

1) !(c = 0 && 0 <= c <= 100)
2) !(c = 0 && c = 100)
3) (constraint (<= (function 0) 0)) 
   (constraint (>= (function 0) 100))
4) (constraint (<= (function 100) 0)) 
   (constraint (>= (function 100) 100))
5) (constraint (<= (function x) 100))
   (constraint (>= (function x) 0)) 
   (constraint (> x 0))
   (constraint (<= x 100))
6) (constraint (<= (function x) 100))
   (constraint (>= (function x) 0)) 
   (constraint (>= x 0))
   (constraint (< x 100))
